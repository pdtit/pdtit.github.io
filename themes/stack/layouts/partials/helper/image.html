/// Params: /// Resources: Where to search for images /// Image: Image URL /// Returns: /// Local: true if the image is local, false if it is remote /// Resource: Hugo image resource object, not nil if the image is local /// Permalink: Image URL /// Height: Image height /// Width: Image width {{ $result := "" }} {{ if .Image }} {{ $url := urls.Parse .Image }} {{ $resource := "" }} {{ if not (in (slice "https" "http") $url.Scheme) }} {{ $resource = .Resources.Get (printf "%s" (.Image | safeURL)) }} {{ else }} {{/* Hugo 0.128.2 compatibility: Skip remote resource processing, just use URL */}} {{/* Remote images will use the URL directly without local caching */}} {{ end }} {{ if $resource }} {{- $isSVG := eq $resource.MediaType.SubType "svg" -}} {{- if $isSVG -}} {{ $result = dict "Local" true "Resource" $resource "Permalink" $resource.RelPermalink "Height" nil "Width" nil }} {{- else -}} {{- $result = dict "Local" true "Resource" $resource "Permalink" $resource.RelPermalink "Height" $resource.Height "Width" $resource.Width }} {{- end -}} {{ else }} {{/* No resource found - return URL-only dict for remote/external images */}} {{ $result = dict "Local" false "Resource" nil "Permalink" .Image "Height" nil "Width" nil }} {{ end }} {{ end }} {{ return $result }}
